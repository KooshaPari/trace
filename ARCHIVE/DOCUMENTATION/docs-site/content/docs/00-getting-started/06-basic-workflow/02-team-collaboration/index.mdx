
## Usage

### Basic Usage

```typescript
import { TraceRTM } from '@tracertm/core'

const tracertm = new TraceRTM()
const result = await tracertm.teamcollaboration()
```

### Advanced Usage

```typescript
const config = {
  team_collaboration: {
    enabled: true,
    options: {
      mode: 'advanced'
    }
  }
}

const tracertm = new TraceRTM(config)
```
