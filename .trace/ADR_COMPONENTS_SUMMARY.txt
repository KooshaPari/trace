================================================================================
                            USAGE EXAMPLES
================================================================================

IMPORT:
  import {
    ADRCard,
    ADREditor,
    ADRTimeline,
    ADRGraph,
    DecisionMatrix,
    ComplianceGauge,
  } from "@/components/specifications/adr";

BASIC DISPLAY:
  <ADRCard
    adr={adrData}
    onClick={() => handleSelect(adrData)}
    showComplianceGauge={true}
  />

TIMELINE VIEW:
  <ADRTimeline
    adrs={adrList}
    onADRClick={(adr) => handleSelect(adr)}
  />

GRAPH VISUALIZATION:
  <ADRGraph
    adrs={adrList}
    selectedAdrId={selectedId}
    onAdrSelect={(adr) => handleSelect(adr)}
  />

FORM EDITING:
  <ADREditor
    initialData={existingAdr}
    onSave={async (data) => await saveAdr(data)}
    onCancel={() => setEditMode(false)}
  />
